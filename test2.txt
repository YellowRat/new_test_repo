public class MergeSortAlgorithm {

    /**
     * The main recursive function that sorts the array A[low..high] using Merge Sort.
     */
    public static void mergeSort(int[] A, int low, int high) {
        // Base case: If low is less than high, there is more than one element.
        if (low < high) {
            // Find the middle point
            int mid = low + (high - low) / 2;

            // Recursively sort the first half
            mergeSort(A, low, mid);

            // Recursively sort the second half
            mergeSort(A, mid + 1, high);

            // Merge the sorted halves
            merge(A, low, mid, high);
        }
    }

    /**
     * Merges two subarrays of A[].
     * The first subarray is A[low..mid].
     * The second subarray is A[mid+1..high].
     */
    public static void merge(int[] A, int low, int mid, int high) {
        // Find the sizes of the two subarrays to be merged
        int n1 = mid - low + 1;
        int n2 = high - mid;

        // Create temporary arrays
        int[] L = new int[n1];
        int[] R = new int[n2];

        // Copy data to temp arrays L[] and R[]
        for (int i = 0; i < n1; ++i)
            L[i] = A[low + i];
        for (int j = 0; j < n2; ++j)
            R[j] = A[mid + 1 + j];

        // Initial indexes of first, second, and merged subarrays
        int i = 0, j = 0;
        int k = low; // k is the current index in the main array A

        // Merge the temp arrays back into A[low..high]
        while (i < n1 && j < n2) {
            if (L[i] <= R[j]) {
                A[k] = L[i];
                i++;
            } else {
                A[k] = R[j];
                j++;
            }
            k++;
        }

        // Copy remaining elements of L[] if any
        while (i < n1) {
            A[k] = L[i];
            i++;
            k++;
        }

        // Copy remaining elements of R[] if any
        while (j < n2) {
            A[k] = R[j];
            j++;
            k++;
        }
    }

    /**
     * Utility function to print the array.
     */
    public static void printArray(int[] A) {
        for (int i = 0; i < A.length; i++) {
            System.out.print(A[i] + " ");
        }
        System.out.println();
    }

    // Main method to test the algorithm
    public static void main(String[] args) {
        int[] data = { 12, 11, 13, 5, 6, 7 };
        System.out.print("Given Array: ");
        printArray(data);

        // Call mergeSort
        mergeSort(data, 0, data.length - 1);

        System.out.print("Sorted Array: ");
        printArray(data);
    }
